	static void Parsing(String text1) //parçalama metodu
	{ 
		StringTokenizer token=new StringTokenizer(text1); //parsing
		System.out.println("////////////////////////PARSING///////////////////////////");
		while(token.hasMoreTokens()) 
		{
			System.out.println(token.nextToken());
	    }	
	}
	
	public denemegui() { 
		setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		setBounds(100, 100,540, 312);
		setResizable(false);
		contentPane = new JPanel();
		contentPane.setBorder(new EmptyBorder(5, 5, 5, 5));
		setContentPane(contentPane);
		
		JButton buton1 = new JButton("resim ekle");
		buton1.setBounds(10, 101, 96, 37);
		buton1.addActionListener( new ActionListener() {//buton1(resim seçme)in yapacaðý görev
			
			public void actionPerformed(ActionEvent e) {
				
				 int state = dosya.showOpenDialog(null);//dosya seçme ekraný açýlýr
		         File file = dosya.getSelectedFile(); //dosya seçimi yapýlýr
		         String s="ÝPTAL EDÝLDÝ";

		         if(file != null && state == JFileChooser.APPROVE_OPTION) //dosya secilirse
		         {
		           s = "DOSYA SEÇÝLDÝ: " + file.getPath();// bu mesaj gösterilir
		         }
		         else JOptionPane.showMessageDialog(null,s);//dosya seçilmezse bu mesaj gösterilir
 
		         Tesseract tesseract = new Tesseract();
				 try 
				 {
					tesseract.setDatapath("C:\\OCR Tesseract\\Tess4J\\tessdata");
					String text = tesseract.doOCR(new File(file.getPath()));
					textArea.setText(text);
				 } 
				 
				 catch (TesseractException e1)
				 {	
				   e1.printStackTrace();
				 }
				     
		 }});
		contentPane.setLayout(null);
		contentPane.add(buton1);
		
		scrollPane = new JScrollPane();
		scrollPane.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_ALWAYS);
		scrollPane.setHorizontalScrollBarPolicy(ScrollPaneConstants.HORIZONTAL_SCROLLBAR_ALWAYS);
		scrollPane.setBounds(110, 11, 359, 226);
		contentPane.add(scrollPane);
		textArea = new JTextArea();
		scrollPane.setViewportView(textArea);
	}
}
