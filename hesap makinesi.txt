package ramazandeneme;
import java.util.*;

public class try2 {

	public static void main(String[] args) {

		Scanner s=new Scanner(System.in);
		int a=s.nextInt();
		int b=s.nextInt();
		System.out.println("carpýmlarý:"+carp(a,b));
		System.out.println("toplamlarý:"+topla(a,b));
		System.out.println("bolumleri:"+bol(a,b));
		System.out.println("farklari:"+cýkar(a,b));
	}

	static int carp(int a1,int b1){
		return a1*b1;
	}
	static int topla(int a1,int b1){
		return a1+b1;
	}
	static float bol(float a1,float b1){
		return a1/b1;
	}
	static int cýkar(int a1,int b1){
		return a1-b1;
	}
	
}
